# Slim runtime dependencies for the Streamlit dashboard ONLY.
#
# Streamlit Cloud's resolver picks the dependency file in the entrypoint's
# directory (this one) over the project-root requirements.txt — saving ~3 min
# of cold-start install time by skipping dbt-core and its 30+ transitive deps
# that the dashboard never imports at runtime.
#
# Pins are deliberately loose (>=) so that whatever Python version Streamlit
# Cloud happens to use (currently Python 3.14), pip resolves to a version
# with prebuilt wheels — avoiding the cmake-required source build that broke
# our first deploy attempt with strict pandas==2.2.3 / pyarrow==18.0.0 pins
# (no cp314 wheels existed for those exact versions).

streamlit>=1.57
duckdb>=1.4
pandas>=2.3
plotly>=5.24
